5 # triangulation
6 #
7 # This file contains functions that can be used to determine the position
8 # of a sound in a mesh network given a set of nodeEvents with associated
9 # information. The nodeEvent (at this point) is a simple struct containing
10 # its x and y position, the confidence of the sound that was just detected,
11 # and the sound pressure level of the sound when it was detected. Using this
12 # information we can create a probability function that can be maximized to
13 # determine the position of the sample sound. The sample sound (for localization
14 # given that the sound has been accurately detected by a set of nodeEvents)
15 # only needs to contain the reference sound pressure level and a reference
16 # distance from which this measurement has been taken.

102 #
103 #
104 # Uses the relative time differences between the node events mixed with
105 # the confidence of sound recognition to determine the standard deviation
106 # of the node event. Node events that are closest to the node event that has
107 # the largest time stamps will have a lower standard deviation. Also, node
108 # events that have a larger sound recognition confidence will be given lower
109 # standard deviations.
110 #
111 # @param nodeEvents The list of associated data when a node detects with some
112 # confidence that the sound has been identified
113 #
114 #
115 def set_node_events_std(node_events):
116 
117  if len(node_events) == 0:
118  raise ValueError("Node event list is of length 0")
119 
120  max_time = 0
121  min_time = node_events[0].get_timestamp()
122  for node_event in node_events:
123  if node_event.get_timestamp() > max_time:
124  max_time = node_event.get_timestamp()
125  elif node_event.get_timestamp() < min_time:
126  min_time = node_event.get_timestamp()
127 
128  for node_event in node_events:
129  time_error = 1.0 - (
130  (max_time - node_event.get_timestamp()) /
131  (max_time - min_time)
132  )
133  node_event.set_std(2 * K / (node_event.confidence + time_error))

138 #
139 # Evaluation function to deterimine with some weight, where a sound is
140 # located in the mesh network of nodes. Given an x and y, this returns
141 # a weight. The higher the weight, the higher the likelihood that the
142 # sound originated from x and y.
143 #
144 # @param x the horizontal location of the node event when the node event
145 # was captured
146 #
147 # @param y the vertical location of the node event when the node event
148 # was captured
149 #
150 # @param rRef The reference distance at which the reference sound
151 # pressure level was recorded
152 #
153 # @param lRef The reference sound pressure level used to determine the
154 # distance from the newly measured sound pressure level
155 #
156 # @param nodeEvents The list of associated data when a node detects with some
157 # confidence that the sound has been identified
158 #
159 # @return The a result, given independent variables, x and y, and a
160 # configuration of node events with per sample sound constants, rRef and
161 # lRef, will be a value representing the likelihood of the sample sound
162 # being located at position (x, y).
163 #
164 #
165 def position_evaluation(x, y, r_ref, l_ref, node_events):
166 
167  set_node_events_std(node_events)
168 
169  return sum(
170  [
171  normal_distribution(
172  (
173  distance_from_detection_event(x, y, n) -
174  distance_from_sound(r_ref, l_ref, n.spl)
175  ) / n.get_std()
176  ) / n.get_std() for n in node_events
177  ]
178  )

220 #
221 # Determines a list of possible positions of where the sound will be
222 # located. These positions are determined by changing iterating through
223 # the list of node events and optimizing the probability density function
224 # with the initial guess being the position of the node event. *Hopefully*
225 # the optimization will find the local minima.
226 #
227 # @param rRef The reference distance at which the reference sound
228 # pressure level was recorded
229 #
230 # @param lRef The reference sound pressure level used to determine the
231 # distance from the newly measured sound pressure level
232 #
233 # @param initGuess The initial guess for gradient decent
234 #
235 # @param nodeEvents The list ofassociated data when a node detects with some
236 # confidence that the sound has been identified
237 #
238 # @return A list of the possible x and y positions of the sound.
239 #
240 #
241 def determine_sound_position_list(r_ref, l_ref, node_events, **kwargs):
242 
243  p_func = lambda v: -1 * position_probability(
244  v[0], v[-1],
245  r_ref, l_ref,
246  node_events
247  )
248 
249  max_list = [
250  opt.fmin(p_func, ne.get_pos(), full_output=1, **kwargs)
251  for ne in node_events
252  ]
253 
254  max_vals = [
255  (Point(x, y), -z) for (x, y), z, _, _, _ in max_list
256  ]
257 
258  return max_vals

263 #
264 # Given a list of "optimized" points and their corresponding probabilities
265 # and a list of labels returned from the clustering algorithm, this
266 # function goes through all of the optimized points and returns the ones with
267 # the highest probabilities and issues them as cluter centers. This function
268 # is used to ensure that the center for each cluster also has the highest
269 # probability in the cluster of being the sound position.
270 #
271 # @param opt_vals A list of tuples where each element is a key-value pair
272 # where the key is the Point object of the optimization x and y position
273 # and the value is the associated probability
274 #
275 # @param label_list A list of integers where the index of the list
276 # corresponds to a certain key-value pair in the first parameter
277 # list and the integer value represents which cluster it belongs to.
278 #
279 #
280 def determine_peaks(opt_vals, label_list):
281  max_prob_list = list()
282  max_point_list = list()
283  for i, (point, prob) in zip(label_list, opt_vals):
284  try:
285  if max_prob_list[i] < prob:
286  max_point_list[i] = point
287  max_prob_list[i] = prob
288  except IndexError:
289  max_point_list.append(point)
290  max_prob_list.append(prob)
291 
292  return max_point_list
